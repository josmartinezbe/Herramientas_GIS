$path   = 'D:\6_MAPAS'
$suffix = ' en el marco de la MLA'

  Where-Object { $_.BaseName -notlike "*$suffix" } |
  ForEach-Object {

    Write-Host "[SIMULACIÓN] $($_.Name) -> $newName"
  }
